-- 可选：提供便捷索引函数
function ChemDB.buildIndexById()
    local idx = {}

    for _, v in ipairs(ChemDB.inorganicCompounds) do
        idx[v.id] = v
    end

    local org = ChemDB.organicCompounds
    for _, group in pairs(org) do
        if type(group) == "table" then
            for _, v in ipairs(group) do
                if type(v) == "table" and v.id then
                    idx[v.id] = v
                end
            end
        end
    end

    if ChemDB.massiveLibrary then
        for _, section in pairs(ChemDB.massiveLibrary) do
            if type(section) == "table" then
                for _, v in ipairs(section) do
                    if type(v) == "table" and v.id then
                        idx[v.id] = v
                    end
                end
            end
        end
    end

    return idx
end

function ChemDB.findByAlias(name)
    local id = ChemDB.aliases[name]
    if not id then return nil end
    local idx = ChemDB.buildIndexById()
    return idx[id]
end

return ChemDB
